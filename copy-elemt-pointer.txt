#include<stdio.h>
int main()
{
    int n,i;
    printf("enter a size\n");
    scanf("%d",&n);
    int p[n],q[n];
    int *ptr=&p;
    int *ptr1=&q;
    printf("enter a array elements\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",(ptr+i));
    }
    printf("array elements are\n");
    for(i=0;i<n;i++)
    {
        printf("%d\n",*(ptr+i));
    }
    printf("copied array are\n");
    for(i=0;i<n;i++)
    {
        *(ptr1+i)=*(ptr+i);
        printf("%d\n",*(ptr1+i));
    }
}
