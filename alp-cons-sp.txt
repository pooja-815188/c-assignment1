#include<stdio.h>
#include<string.h>
int main()
{
    char str[20];
    printf("enter a string\n");
    scanf("%s",str);
    int i=0,alphabet=0,digit=0,other=0;
    while(str[i]!='\0')
    {
    if(str[i]>='a' && str[i]<='z' || str[i]>='A' && str[i]<='Z')
    {
        alphabet++;
    }
    else if(str[i]>='0' && str[i]<='9')
    {
        digit++;
    }
    else
    {
        other++;
    }
    i++;
    }
    printf("alphabet=%d\n",alphabet);
    printf("digit=%d\n",digit);
    printf("other=%d\n",other);
}
