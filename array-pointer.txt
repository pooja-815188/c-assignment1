#include<stdio.h>
int main()
{
    int n,i;
    printf("enter a size\n");
    scanf("%d",&n);
    int arr[n];
    int *ptr=arr;
    printf("enter a array elemts\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",(ptr+i));
    }
    printf("array elemts are\n");
    for(i=0;i<n;i++)
    {
        printf("%d\n",*(ptr+i));
    }
    return 0;
}
